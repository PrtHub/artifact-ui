## Usage

```tsx
import ClimateArtboard from "@/components/ui/climate-artboard";

export default function WeatherApp() {
  return (
    <ClimateArtboard
      title="Welcome to Weather App"
      subtitle="Your local weather forecast"
      theme="glass"
      autoAnimate={true}
      showDetails={true}
      apiKey={process.env.NEXT_PUBLIC_WEATHER_API_KEY}
      onWeatherChange={(weather) => {
        console.log("Weather changed:", weather);
      }}
    />
  );
}
```

## Props

| Name              | Type                     | Description                                                       |
| ----------------- | ------------------------ | ----------------------------------------------------------------- |
| `title`           | `string`                 | The title of the weather app.                                     |
| `subtitle`        | `string`                 | The subtitle of the weather app.                                  |
| `theme`           | `string`                 | The theme of the weather app.                                     |
| `apiKey`          | `string`                 | The API key for the weather data.                                 |
| `onWeatherChange` | `(weather: any) => void` | A callback function that is called when the weather data changes. |
| `autoAnimate`     | `boolean`                | Whether to automatically animate the weather app.                 |
| `showDetails`     | `boolean`                | Whether to show the detailed weather information.                 |
