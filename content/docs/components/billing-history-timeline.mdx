## Usage

```tsx
import {
  BillingHistoryTimeline,
  BillingEvent,
} from "@/components/ui-canvas/billing-history-timeline";

const events: BillingEvent[] = [
  {
    date: "2024-01-20",
    type: "payment",
    status: "completed",
    amount: 199.99,
    description: "Premium Plan Payment",
    details: "Successfully processed monthly subscription payment",
  },
  {
    date: "2024-01-15",
    type: "usage",
    description: "Usage Spike Detected",
    details: "API calls increased by 250%",
  },
];

export default function MyComponent() {
  return <BillingHistoryTimeline events={events} variant="gradient" />;
}
```

## Props

### BillingHistoryTimeline

| Prop      | Type                                                                         | Default     | Description                        |
| --------- | ---------------------------------------------------------------------------- | ----------- | ---------------------------------- |
| events    | `BillingEvent[]`                                                             | -           | An array of billing events         |
| variant   | `"default" \| "minimal" \| "gradient" \| "neon" \| "watercolor" \| "sketch"` | `"default"` | Visual style variant               |
| className | `string`                                                                     | -           | Additional CSS classes for styling |

### BillingEvent

| Prop        | Type                                                  | Default     | Description                          |
| ----------- | ----------------------------------------------------- | ----------- | ------------------------------------ |
| date        | `string`                                              | -           | The date of the billing event        |
| type        | `"payment" \| "invoice" \| "usage" \| "subscription"` | `"default"` | The type of billing event            |
| status      | `"pending" \| "completed"`                            | `"default"` | The status of the billing event      |
| amount      | `number`                                              | -           | The amount of the billing event      |
| description | `string`                                              | -           | The description of the billing event |
| details     | `string`                                              | -           | Additional details about the event   |
