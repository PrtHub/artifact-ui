## Usage

```tsx
import { Alert } from "@/components/ui/ink-alert";

const AlertDemo = () => {
  return (
    <Alert
      variant="default"
      position="top-right"
      duration={5000}
      onClose={() => {}}
    >
      Your artwork has been saved successfully.
    </Alert>
  );
};

export default AlertDemo;
```

## Examples

### Variants

```tsx
import { Alert } from "@/components/ui/ink-alert";

export function AlertVariants() {
  return (
    <>
      <Alert variant="default">
        A default ink alert with a blue gradient effect.
      </Alert>

      <Alert variant="destructive">
        A destructive ink alert with a red gradient effect.
      </Alert>

      <Alert variant="success">
        A success ink alert with a green gradient effect.
      </Alert>

      <Alert variant="warning">
        A warning ink alert with a yellow gradient effect.
      </Alert>
    </>
  );
}
```

### Positioning

```tsx
import { Alert } from "@/components/ui/ink-alert";

export function AlertPositions() {
  return (
    <>
      <Alert position="top-right">Appears in the top-right corner</Alert>

      <Alert position="bottom-center">Appears in the bottom-center</Alert>
    </>
  );
}
```

### Auto-dismiss

```tsx
import { Alert } from "@/components/ui/ink-alert";

export function AutoDismissAlert() {
  return (
    <Alert variant="success" duration={5000} position="top-right">
      This alert will automatically dismiss after 5 seconds
    </Alert>
  );
}
```

### Dismissible

```tsx
import { Alert } from "@/components/ui/ink-alert";

export function DismissibleAlert() {
  return (
    <Alert variant="default" onClose={() => console.log("Alert closed")}>
      Click the X button to dismiss this alert with a smooth exit animation.
    </Alert>
  );
}
```

## Props

| Prop name | Type                                                                                            | Default     | Description                            |
| --------- | ----------------------------------------------------------------------------------------------- | ----------- | -------------------------------------- |
| variant   | "default" \| "destructive" \| "success" \| "warning"                                            | "default"   | The style variant of the alert         |
| position  | "top-right" \| "top-left" \| "bottom-right" \| "bottom-left" \| "top-center" \| "bottom-center" | "top-right" | Position of the alert on screen        |
| duration  | number                                                                                          | undefined   | Auto-dismiss duration in milliseconds  |
| onClose   | () => void                                                                                      | undefined   | Callback function when alert is closed |
| children  | React.ReactNode                                                                                 | -           | The content of the alert               |
| className | string                                                                                          | -           | Additional CSS classes                 |

## Features

- **Modern Design**: Sleek gradients and backdrop blur effects
- **Flexible Positioning**: Six different positions for alert placement
- **Smooth Animations**: Position-based enter/exit animations
- **Auto-dismiss**: Optional automatic dismissal after specified duration
- **Customizable**: Supports custom styling and multiple variants
- **Accessible**: Built with proper ARIA attributes and keyboard support
