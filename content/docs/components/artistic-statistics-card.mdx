## Usage

```tsx
import { ArtisticStatisticsCard } from "@/components/ui/artistic-statistics-card";
import { Users } from "lucide-react";

export default function Statistics() {
  return (
    <ArtisticStatisticsCard
      title="Active Users"
      value="14,532"
      description="Real-time platform users"
      icon={<Users />}
      variant="glass"
      animation="glow"
      trend={8.2}
    />
  );
}
```

## Props

| Prop        | Type                                                      | Description                               | Default   |
| ----------- | --------------------------------------------------------- | ----------------------------------------- | --------- |
| title       | string                                                    | The title of the statistics card          | Required  |
| value       | string \| number                                          | The main value to display                 | Required  |
| description | string                                                    | Optional description text                 | undefined |
| icon        | ReactNode                                                 | Optional icon component                   | undefined |
| variant     | 'default' \| 'gradient' \| 'outline' \| 'glass' \| 'neon' | Visual style variant                      | 'default' |
| animation   | 'none' \| 'pulse' \| 'bounce' \| 'float' \| 'glow'        | Animation style                           | 'none'    |
| trend       | number                                                    | Percentage change to show trend indicator | undefined |
| className   | string                                                    | Additional CSS classes                    | undefined |

## Examples

```tsx
<ArtisticStatisticsCard
```


## Customization

### Custom Styling

You can customize the appearance using the `className` prop:

```tsx
<ArtisticStatisticsCard
  className="bg-[#1a1a1a] text-white"
  title="Custom Style"
  value="42"
/>
```

### Responsive Layout

The cards are designed to work in grid layouts:

```tsx
<div className="grid gap-6 sm:grid-cols-2">
  <ArtisticStatisticsCard ... />
  <ArtisticStatisticsCard ... />
</div>
```
