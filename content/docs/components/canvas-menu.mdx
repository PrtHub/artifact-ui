## Installation

<Steps>

<Step>Install the required dependencies.</Step>

```bash
npm install @radix-ui/react-menubar
```

<Step>Copy and paste the following code into your project.</Step>

`components/ui/canvas-menu.tsx`

<ComponentSource name="canvas-menu" />

<Step>Update the import paths to match your project setup.</Step>

```tsx
import {
  CanvasMenu,
  CanvasMenuMenu,
  CanvasMenuTrigger,
  CanvasMenuContent,
  CanvasMenuItem,
  CanvasMenuSeparator,
  CanvasMenuCheckboxItem,
  CanvasMenuRadioGroup,
  CanvasMenuRadioItem,
  CanvasMenuSub,
  CanvasMenuSubTrigger,
  CanvasMenuSubContent,
  CanvasMenuLabel,
} from "@/components/ui/canvas-menu";
```

</Steps>

## Usage

### Basic Usage

```tsx
export default function ArtistTools() {
  const [showGrid, setShowGrid] = React.useState(true);
  const [artStyle, setArtStyle] = React.useState("modern");

  return (
    <CanvasMenu>
      <CanvasMenuMenu>
        <CanvasMenuTrigger>Artist Menu</CanvasMenuTrigger>
        <CanvasMenuContent>
          {/* File Operations */}
          <CanvasMenuItem>
            New Canvas{" "}
            <span className="ml-auto text-xs text-muted-foreground">⌘N</span>
          </CanvasMenuItem>
          <CanvasMenuItem>
            Open...{" "}
            <span className="ml-auto text-xs text-muted-foreground">⌘O</span>
          </CanvasMenuItem>
          <CanvasMenuSeparator />

          {/* Tools Submenu */}
          <CanvasMenuSub>
            <CanvasMenuSubTrigger>Art Tools</CanvasMenuSubTrigger>
            <CanvasMenuSubContent>
              <CanvasMenuItem>
                Brush{" "}
                <span className="ml-auto text-xs text-muted-foreground">B</span>
              </CanvasMenuItem>
              <CanvasMenuItem>
                Pencil{" "}
                <span className="ml-auto text-xs text-muted-foreground">P</span>
              </CanvasMenuItem>
            </CanvasMenuSubContent>
          </CanvasMenuSub>

          {/* Settings */}
          <CanvasMenuSeparator />
          <CanvasMenuCheckboxItem
            checked={showGrid}
            onCheckedChange={setShowGrid}
          >
            Show Grid{" "}
            <span className="ml-auto text-xs text-muted-foreground">⌘G</span>
          </CanvasMenuCheckboxItem>

          {/* Art Style Selection */}
          <CanvasMenuSeparator />
          <CanvasMenuLabel>Art Style</CanvasMenuLabel>
          <CanvasMenuRadioGroup value={artStyle} onValueChange={setArtStyle}>
            <CanvasMenuRadioItem value="classic">Classic</CanvasMenuRadioItem>
            <CanvasMenuRadioItem value="modern">Modern</CanvasMenuRadioItem>
          </CanvasMenuRadioGroup>
        </CanvasMenuContent>
      </CanvasMenuMenu>
    </CanvasMenu>
  );
}
```

## Props

### CanvasMenu

| Prop name | Type           | Default | Description                                             |
| --------- | -------------- | ------- | ------------------------------------------------------- |
| modal     | boolean        | true    | Whether to render in a modal context                    |
| dir       | "ltr" \| "rtl" | "ltr"   | Text direction                                          |
| loop      | boolean        | true    | Whether keyboard navigation should loop around items    |
| asChild   | boolean        | false   | Change the root element using Radix UI's `Slot` utility |

### CanvasMenuItem

| Prop name | Type       | Default | Description                                     |
| --------- | ---------- | ------- | ----------------------------------------------- |
| disabled  | boolean    | false   | Whether the item is disabled                    |
| onSelect  | () => void | -       | Callback when item is selected                  |
| className | string     | -       | Additional CSS classes                          |
| asChild   | boolean    | false   | Change the root element using Radix UI's `Slot` |
| textValue | string     | -       | Optional text used for typeahead purposes       |

### CanvasMenuCheckboxItem

| Prop name       | Type                       | Default | Description                                     |
| --------------- | -------------------------- | ------- | ----------------------------------------------- |
| checked         | boolean                    | -       | Whether the item is checked                     |
| onCheckedChange | (checked: boolean) => void | -       | Callback when checked state changes             |
| disabled        | boolean                    | false   | Whether the item is disabled                    |
| className       | string                     | -       | Additional CSS classes                          |
| asChild         | boolean                    | false   | Change the root element using Radix UI's `Slot` |

### CanvasMenuRadioGroup

| Prop name     | Type                    | Default | Description                   |
| ------------- | ----------------------- | ------- | ----------------------------- |
| value         | string                  | -       | The selected value            |
| onValueChange | (value: string) => void | -       | Callback when value changes   |
| disabled      | boolean                 | false   | Whether the group is disabled |
| className     | string                  | -       | Additional CSS classes        |

### CanvasMenuRadioItem

| Prop name | Type    | Default | Description                                     |
| --------- | ------- | ------- | ----------------------------------------------- |
| value     | string  | -       | Value of the radio item                         |
| disabled  | boolean | false   | Whether the item is disabled                    |
| className | string  | -       | Additional CSS classes                          |
| asChild   | boolean | false   | Change the root element using Radix UI's `Slot` |

