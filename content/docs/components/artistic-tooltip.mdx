## Usage

```tsx
import { useState } from "react";
import { ArtisticTooltip } from "@/components/ui/artistic-tooltip";

export default function MyComponent() {
  const [showTooltip, setShowTooltip] = useState(false);

  return (
    <ArtisticTooltip
      content="Your tooltip message"
      show={showTooltip}
      style="watercolor"
    >
      <button
        onMouseEnter={() => setShowTooltip(true)}
        onMouseLeave={() => setShowTooltip(false)}
      >
        Hover me
      </button>
    </ArtisticTooltip>
  );
}
```

## Props

| Prop      | Type                                    | Default        | Description                               |
| --------- | --------------------------------------- | -------------- | ----------------------------------------- |
| content   | `string`                                | -              | The content to display inside the tooltip |
| show      | `boolean`                               | `false`        | Controls the visibility of the tooltip    |
| style     | `"watercolor"` \| `"oil"` \| `"pencil"` | `"watercolor"` | The artistic style of the tooltip         |
| className | `string`                                | -              | Additional CSS classes to apply           |
| children  | `ReactNode`                             | -              | The element that triggers the tooltip     |
