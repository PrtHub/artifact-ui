## Usage

```tsx
import CanvasTimeline from "@/components/ui/canvas-timeline";

interface TimelineStep {
  id: string;
  thumbnail: string;
  timestamp: Date;
  label: string;
  type: "brush" | "eraser" | "shape" | "text" | "layer" | "filter" | "snapshot";
  description?: string;
  duration?: number; // in seconds
  markers?: Array<{
    time: number; // position in seconds
    label: string;
    color?: string;
  }>;
}

interface CanvasTimelineProps {
  steps: TimelineStep[];
  currentStepIndex: number;
  onStepChange?: (index: number) => void;
  onSnapshot?: () => void;
  onUndo?: () => void;
  onRedo?: () => void;
  onSave?: () => void;
  className?: string;
  canUndo?: boolean;
  canRedo?: boolean;
}

export default function YourComponent() {
  const [currentStep, setCurrentStep] = useState(0);

  return (
    <CanvasTimeline
      steps={steps}
      currentStepIndex={currentStep}
      onStepChange={setCurrentStep}
      canUndo={true}
      canRedo={true}
    />
  );
}
```

## Keyboard Shortcuts

- Space: Play/Pause
- Left Arrow: Previous step
- Right Arrow: Next step
- Ctrl/Cmd + Z: Undo
- Ctrl/Cmd + Y: Redo

## Props

| Prop               | Type                      | Description                                          |
| ------------------ | ------------------------- | ---------------------------------------------------- |
| `steps`            | `TimelineStep[]`          | Array of timeline steps with thumbnails and metadata |
| `currentStepIndex` | `number`                  | Current active step index                            |
| `onStepChange`     | `(index: number) => void` | Callback function when step changes                  |
| `onSnapshot`       | `() => void`              | Callback function to handle snapshot creation        |
| `onUndo`           | `() => void`              | Callback function to handle undo action              |
| `onRedo`           | `() => void`              | Callback function to handle redo action              |
| `onSave`           | `() => void`              | Callback function to handle save action              |
| `className`        | `string?`                 | Optional className for styling                       |
| `canUndo`          | `boolean?`                | Optional boolean to indicate if undo is available    |
| `canRedo`          | `boolean?`                | Optional boolean to indicate if redo is available    |
