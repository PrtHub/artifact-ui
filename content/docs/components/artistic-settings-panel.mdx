## Usage

```tsx
import { ArtisticSettingsPanel } from "@/components/ui/artistic-settings-panel";

export default function YourComponent() {
  const handleSettingsChange = (settings) => {
    console.log("Settings changed:", settings);
    // Apply the settings to your content
  };

  return (
    <div>
      <ArtisticSettingsPanel
        defaultOpen={false}
        onSettingsChange={handleSettingsChange}
      />
      {/* Your content here */}
    </div>
  );
}
```

## Props

| Prop             | Type                                 | Default | Description                                  |
| ---------------- | ------------------------------------ | ------- | -------------------------------------------- |
| defaultOpen      | boolean                              | false   | Controls the initial open state of the panel |
| onSettingsChange | (settings: ArtisticSettings) => void | -       | Callback when settings change                |

## Settings Interface

```typescript
interface ArtisticSettings {
  brightness: number; // 0-200%
  contrast: number; // 0-200%
  saturation: number; // 0-200%
  blur: number; // 0-20px
  isDarkMode: boolean;
  isAutoEnhance: boolean;
  preset: "default" | "dramatic" | "vintage" | "dreamy";
}
```

## Keyboard Shortcuts

| Shortcut   | Action       |
| ---------- | ------------ |
| `Ctrl + \` | Toggle panel |
