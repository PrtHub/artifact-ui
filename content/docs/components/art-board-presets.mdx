## Usage

```tsx
import ArtBoardPresets from "@/components/ui/art-board-presets";

export default function YourComponent() {
  return (
    <ArtBoardPresets
      onSelectPreset={(width, height) => {
        // Handle the selected size
        console.log(`Selected size: ${width}Ã—${height}`);
      }}
    />
  );
}
```

## Props

- `onSelectPreset` - Optional callback function that receives the selected width and height when a preset is clicked
