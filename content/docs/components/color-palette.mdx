### Example Implementation

```bash
import { ColorPalette } from "@/components/ui/color-palette";

// Theme switcher example
export function ThemeCustomizer() {
  return (
    <div className="space-y-4">
      <h3>Choose Your Theme Style</h3>
      <ColorPalette
        variant="watercolor"
        onChange={(colors) => updateTheme(colors)}
      />
    </div>
  );
}
```

## Props

### ColorPalette

| Prop name | Type                                                              | Default      | Description                       |
| --------- | ----------------------------------------------------------------- | ------------ | --------------------------------- |
| variant   | "watercolor" \| "oilpaint" \| "charcoal" \| "pastel" \| "acrylic" | "watercolor" | The artistic style of the palette |
| showNames | boolean                                                           | true         | Show color group names            |
| className | string                                                            | -            | Additional CSS classes            |
