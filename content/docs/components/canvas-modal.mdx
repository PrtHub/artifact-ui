## Installation

<Steps>

<Step>Install the following dependencies:</Step>

```bash
npm install @radix-ui/react-dialog motion/react lucide-react
```

<Step>Copy and paste the following code into your project.</Step>

`components/ui/canvas-modal.tsx`

<ComponentSource name="canvas-modal" />

<Step>Update the import paths to match your project setup.</Step>

```tsx
import {
  CanvasModal,
  CanvasModalContent,
  CanvasModalDescription,
  CanvasModalHeader,
  CanvasModalTitle,
  CanvasModalTrigger,
  CanvasModalFooter,
} from "@/components/ui/canvas-modal";
```

</Steps>

## Examples

### Basic Usage

```tsx
import { Button } from "@/components/ui/button";
import {
  CanvasModal,
  CanvasModalContent,
  CanvasModalHeader,
  CanvasModalTitle,
  CanvasModalTrigger,
} from "@/components/ui/canvas-modal";

export function BasicModal() {
  return (
    <CanvasModal>
      <CanvasModalTrigger asChild>
        <Button variant="outline">Open Modal</Button>
      </CanvasModalTrigger>
      <CanvasModalContent>
        <CanvasModalHeader>
          <CanvasModalTitle>Modal Title</CanvasModalTitle>
        </CanvasModalHeader>
        <div className="py-6">Your content here</div>
      </CanvasModalContent>
    </CanvasModal>
  );
}
```

### With Description and Footer

```tsx
import {
  CanvasModal,
  CanvasModalContent,
  CanvasModalDescription,
  CanvasModalFooter,
  CanvasModalHeader,
  CanvasModalTitle,
  CanvasModalTrigger,
} from "@/components/ui/canvas-modal";

export function ModalWithFooter() {
  return (
    <CanvasModal>
      <CanvasModalTrigger asChild>
        <Button variant="outline">Open Modal</Button>
      </CanvasModalTrigger>
      <CanvasModalContent className="sm:max-w-xl">
        <CanvasModalHeader>
          <CanvasModalTitle>Welcome</CanvasModalTitle>
          <CanvasModalDescription>
            A description of your modal content
          </CanvasModalDescription>
        </CanvasModalHeader>
        <div className="py-6">
          <p className="text-muted-foreground/80">
            Your main content goes here
          </p>
        </div>
        <CanvasModalFooter>
          <Button variant="outline">Cancel</Button>
          <Button>Continue</Button>
        </CanvasModalFooter>
      </CanvasModalContent>
    </CanvasModal>
  );
}
```

### Custom Styling

```tsx
export function CustomStyledModal() {
  return (
    <CanvasModal>
      <CanvasModalTrigger asChild>
        <Button>Custom Modal</Button>
      </CanvasModalTrigger>
      <CanvasModalContent className="bg-gradient-to-br from-primary/5 via-background to-background/95">
        <CanvasModalHeader className="border-b border-primary/10 pb-4">
          <CanvasModalTitle className="bg-gradient-to-r from-primary to-primary/60 bg-clip-text text-transparent">
            Custom Styled Modal
          </CanvasModalTitle>
        </CanvasModalHeader>
        <div className="py-6">
          <p className="text-muted-foreground/80">
            Modal with custom gradient background and styled title
          </p>
        </div>
      </CanvasModalContent>
    </CanvasModal>
  );
}
```

## Props

### CanvasModal

| Prop         | Type                    | Default | Description                        |
| ------------ | ----------------------- | ------- | ---------------------------------- |
| open         | boolean                 | -       | Controlled open state of the modal |
| onOpenChange | (open: boolean) => void | -       | Callback when open state changes   |
| children     | React.ReactNode         | -       | The content of the modal           |

### CanvasModalTrigger

| Prop     | Type            | Default | Description                                   |
| -------- | --------------- | ------- | --------------------------------------------- |
| asChild  | boolean         | false   | Whether to merge props onto the child element |
| children | React.ReactNode | -       | The trigger element that opens the modal      |

### CanvasModalContent

| Prop      | Type            | Default | Description                             |
| --------- | --------------- | ------- | --------------------------------------- |
| className | string          | -       | Additional CSS classes for styling      |
| children  | React.ReactNode | -       | The content to display inside the modal |

### CanvasModalHeader

| Prop      | Type            | Default | Description                                  |
| --------- | --------------- | ------- | -------------------------------------------- |
| className | string          | -       | Additional CSS classes for styling           |
| children  | React.ReactNode | -       | Header content (typically Title/Description) |

### CanvasModalTitle

| Prop      | Type            | Default | Description                        |
| --------- | --------------- | ------- | ---------------------------------- |
| className | string          | -       | Additional CSS classes for styling |
| children  | React.ReactNode | -       | The title text                     |

### CanvasModalDescription

| Prop      | Type            | Default | Description                        |
| --------- | --------------- | ------- | ---------------------------------- |
| className | string          | -       | Additional CSS classes for styling |
| children  | React.ReactNode | -       | The description text               |

### CanvasModalFooter

| Prop      | Type            | Default | Description                               |
| --------- | --------------- | ------- | ----------------------------------------- |
| className | string          | -       | Additional CSS classes for styling        |
| children  | React.ReactNode | -       | Footer content (typically action buttons) |

The Canvas Modal is built on top of [Radix UI Dialog](https://www.radix-ui.com/primitives/docs/components/dialog) and follows the WAI-ARIA Dialog Pattern for accessibility.
